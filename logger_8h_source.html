100 template<typename... Args>
101 inline void log (Severity s, fmtlib::format_string<Args...> fmt, Args && ... args) const {
102 if (isEnabled (s)) {
103 const auto msg { fmtlib::format (fmt, std::forward<Args>(args)...) };
104 const auto ts { std::chrono::duration_cast<std::chrono::milliseconds> (
105 std::chrono::system_clock::now().time_since_epoch()
106 ) };
107
108 for (const auto &t: _transport)
109 std::visit ([ &msg, s, ts ] (const auto &t) { t.log (msg, s, ts); }, t);
110 }
111 }
