99 template<typename... Args>
100 inline void log (Severity s, fmtlib::format_string<Args...> fmt, Args && ... args) const {
101 if (isEnabled (s)) {
102 const auto msg { fmtlib::format (fmt, std::forward<Args>(args)...) };
103 const auto ts { std::chrono::duration_cast<std::chrono::milliseconds> (
104 std::chrono::system_clock::now().time_since_epoch()
105 ) };
106
107 for (const auto &t: _transport)
108 std::visit ([ &msg, s, ts ] (const auto &t) { t.log (msg, s, ts); }, t);
109 }
110 }
